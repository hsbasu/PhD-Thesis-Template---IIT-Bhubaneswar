
\defbibheading{subbibliography}{\section{#1}}
\defbibfilter{booksandchapters}{%
( type=book or type=incollection )
}
\defbibfilter{papers}{%
 ( type=article or type=unpublished )
}
\defbibfilter{conferences}{%
 ( type=inproceedings )
}

\pagestyle{fancy}
\fancyhf{}
\fancyhead[RE]{\bfseries Publications}
\fancyhead[LO]{\bfseries Publications}
\fancyhead[LE,RO]{\thepage}

\chapter*{Publications}
%% Add the BibTeX entries to the file pub.bib
%% for your own journal articles, books, chapters
%% and proceedings etc. They will be automatically
%% sorted and added to the thesis by the
%% following refsection

\begin{refsection}[110_Publications/pub.bib]
    % Since, with the chapter numbers, the table of contents
    % look ugly, the following four lines remove the chapter
    % numbers leaving only the section number
    \addtocontents{toc}{\protect\renewcommand{\protect\cftchappresnum}{}}
    \renewcommand{\chaptername}{Chapter}
    \renewcommand{\thesection}{\arabic{section}}
    \setcounter{section}{0}
    
    \addcontentsline{toc}{chapter}{Publications}
    \nocite{*}
    %% Uncomment the following line to add journal articles
    \printbibliography[heading={subbibliography},title={Journal Articles},filter={papers},resetnumbers=true]
    
    %% Uncomment the following line to add Conference Proceedings
    \printbibliography[heading={subbibliography},title={Conference Proceedings},filter={conferences},resetnumbers=true]
    
    %% Uncomment the following line to add books and chapters
    \printbibliography[heading={subbibliography},title={Books and Chapters},filter={booksandchapters},resetnumbers=true]
\end{refsection}
